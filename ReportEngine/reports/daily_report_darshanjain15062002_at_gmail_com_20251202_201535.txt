        // Function to generate PDF
        function generatePDF() {
            const { jsPDF } = window.jspdf;

            const doc = new jsPDF();
            doc.text("Daily Portfolio Report", 20, 10);

            // Function to add content to PDF
            function addContent(element, startY) {
                let y = startY;
                const section = document.getElementById(element);
                if (section) {
                    doc.setFontSize(16);
                    doc.text(section.querySelector('h2').innerText, 20, y);
                    y += 10;
                    doc.setFontSize(12);

                    const paragraphs = section.querySelectorAll('p');
                    paragraphs.forEach(p => {
                        const text = doc.splitTextToSize(p.innerText, 170);
                        text.forEach(line => {
                            doc.text(line, 20, y);
                            y += 5;
                        });
                    });

                    const tables = section.querySelectorAll('table');
                    tables.forEach(table => {
                        const headers = Array.from(table.querySelectorAll('th')).map(th => th.innerText);
                        const rows = Array.from(table.querySelectorAll('tr')).slice(1).map(row => {
                            return Array.from(row.querySelectorAll('td')).map(td => td.innerText);
                        });

                        doc.table(20, y, rows, headers, { autoSize: false, columnWidth: 'auto' });
                        y += (rows.length + 2) * 10;
                    });
                }
                return y;
            }

            let currentY = 20;
            currentY = addContent('summary', currentY);
            currentY = addContent('holdings', currentY);
            currentY = addContent('profitloss', currentY);
            currentY = addContent('recommendations', currentY);
            currentY = addContent('forum', currentY);
            currentY = addContent('appendix', currentY);

            doc.save("portfolio_report.pdf");
        }
